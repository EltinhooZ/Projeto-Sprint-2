create database projeto;
use projeto;

create table empresa(
cnpj int primary key,
endereco varchar (60),
telefone varchar (11),
email varchar (50),
cod_verificacao int
);

create table usuario(
idUsuario int primary key auto_increment,
email varchar (50),
senha char (12)
) auto_increment = 1 ;

create table setor(
idSetor int primary key auto_increment,
descricao varchar(50)
) auto_increment = 10 ;

create table sensor(
idSensor int primary key auto_increment,
data_sensor date,
horario time,
lumen int
) auto_increment = 10 ;

alter table empresa add column fkUsuario int;
alter table empresa add foreign key (fkUsuario) references usuario(idUsuario);

select * from empresa, usuario where fkUsuario=idUsuario;

alter table empresa add column fkSetor int;
alter table empresa add foreign key (fkSetor) references setor(idSetor);

select * from empresa, setor where fkSetor=idSetor;

alter table setor add column fkSensor int;
alter table setor add foreign key (fkSensor) references sensor(idSensor);

select * from setor, sensor where fkSensor=idSensor;

alter table sensor add column fkEmpresa int;
alter table sensor add foreign key (fkEmpresa) references empresa(cnpj);

select * from sensor, empresa where fkEmpresa=cnpj;

select * from empresa, usuario, setor, sensor;
